PREFIX ais: <http://example.org/ais#>

SELECT DISTINCT ?mmsi ?name
WHERE {
  ?vessel a ais:Vessel .
  ?vessel ais:mmsi ?mmsi .
  ?vessel ais:name ?name .
}
ORDER BY ?mmsi
